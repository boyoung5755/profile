CREATE TABLE `GALLERY` (
    `IMG_NO` INT NOT NULL COMMENT '이미지번호',
    `IMG_RDATE` DATETIME NOT NULL COMMENT '이미지등록일',
    `IMG_DEL` CHAR(1) NOT NULL DEFAULT 'N' COMMENT 'N:게시 , Y:비공개',
    PRIMARY KEY (`IMG_NO`)
);

CREATE TABLE `GUESTBOOK` (
    `GB_NO` INT NOT NULL COMMENT '방명록번호',
    `GB_RDATE` DATETIME NOT NULL COMMENT '방명록등록일',
    `GB_NICK` VARCHAR(10) NULL COMMENT '닉네임',
    `GB_PW` VARCHAR(15) NOT NULL COMMENT '방명록비밀번호',
    `GB_DEL` CHAR(1) NOT NULL DEFAULT 'N' COMMENT 'N:게시 , Y:비공개',
    `GB_CONTENT` VARCHAR(3999) NOT NULL COMMENT '방명록내용',
    PRIMARY KEY (`GB_NO`)
);

CREATE TABLE `CHAT` (
    `CHAT_NO` INT NOT NULL COMMENT '채팅번호',
    `CHAT_RDATE` DATETIME NOT NULL COMMENT '채팅등록일',
    `CHAT_NICK` VARCHAR(10) NOT NULL COMMENT '채팅닉네임',
    `CHAT_CONTENT` VARCHAR(3999) NOT NULL COMMENT '채팅내용',
    PRIMARY KEY (`CHAT_NO`)
);

CREATE TABLE `STACK` (
    `STACK_NO` INT NOT NULL COMMENT '사용기술번호',
    `STACK_NAME` VARCHAR(100) NOT NULL COMMENT '사용기술 이름',
    PRIMARY KEY (`STACK_NO`)
);

CREATE TABLE `CODE` (
    `CODE_NO` INT NOT NULL COMMENT '코드번호',
    `STACK_NO` INT NOT NULL COMMENT '사용기술번호',
    `CODE_CONTENT` LONGTEXT NOT NULL COMMENT '코드내용',
    PRIMARY KEY (`CODE_NO`)
);
